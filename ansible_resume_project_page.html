Sample Playbook - Install Apache WebLogic

---
- name: Install Oracle WebLogic Server
hosts: weblogic_servers
become: yes
vars:
oracle_home: /u01/app/oracle/middleware
weblogic_version: 12.2.1.4
tasks:
- name: Create Oracle user
user:
name: oracle
state: present
- name: Install Java JDK
yum:
name: java-1.8.0-openjdk-devel
state: present
- name: Copy WebLogic installer
copy:
src: fmw_12.2.1.4.0_wls.jar
dest: /tmp/wls_installer.jar
- name: Run WebLogic silent install
command: java -jar /tmp/wls_installer.jar -silent -responseFile /tmp/wls.rsp
- name: Start WebLogic Admin Server
shell: nohup {{ oracle_home }}/user_projects/domains/base_domain/startWebLogic.sh &
